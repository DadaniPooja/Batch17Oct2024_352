//link section : working with scanf and printf statements
#include <stdio.h>
#include <stdbool.h>
#include <math.h>

//function prototype
bool isPrime(int);
int prime(int);

bool isPrime(int n)
{
	if(n<=1)
		return false;
	if(n==2)
		return true;
	if(n%2==0)
		return false;

	for(int i=3; i<=sqrt(n); i+=2)
	{
		if(n%i==0)
			return false;
	}
	return true;
}

int prime(int n)
{
	int cnt=0;

	for(int i=2; i<=n/2; i++)
	{
		if(isPrime(i) && isPrime(n - i))
		{
			printf("%d = %d + %d\n", n, i, n-i);
			cnt++;
		}
	}

	if (cnt > 0)
		printf("NoofWays = %d\n", cnt);
	else
		printf("NoofWays = -1\n");
}


int main()
{
	int num;
	printf("Enter a positive integer: ");
	while (scanf("%d", &num) != 1 || num <= 0)
	{
		printf("Please enter valid positive integer: ");
		while(getchar() != '\n');
	}
	prime(num);
	return 0;
}
